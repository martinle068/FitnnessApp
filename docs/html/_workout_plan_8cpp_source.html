110void WorkoutPlan::processExercises(const std::string& day, const std::string& dayExercises, std::map<std::string, std::vector<ExerciseDetails>>& weeklyPlan)
111{
112 std::vector<ExerciseDetails> exercises;
113 if (dayExercises.empty() || dayExercises == " ")
114 {
115 weeklyPlan[day] = exercises;
116 return;
117 }
118 std::stringstream ess(dayExercises);
119 std::string exerciseDetails;
120 while (std::getline(ess, exerciseDetails, ';'))
121 {
122 if (exerciseDetails.empty()) continue; // Skip empty entries
123 ExerciseDetails details;
124 std::string exerciseName = exerciseDetails.substr(0, exerciseDetails.find('='));
125 std::string setsReps = exerciseDetails.substr(exerciseDetails.find('=') + 1);
126 details.exerciseName = exerciseName;
127 details.sets = std::stoi(setsReps.substr(0, setsReps.find('x')));
128 details.reps = std::stoi(setsReps.substr(setsReps.find('x') + 1));
129 exercises.push_back(details);
130 }
131 weeklyPlan[day] = exercises;
132}
